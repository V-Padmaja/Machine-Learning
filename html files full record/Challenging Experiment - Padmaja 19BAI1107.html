def adjusted_r2(predictions):
    n = df_scaled.shape[0]
    p = X.shape[1]
    r2 = metrics.r2_score(y_test, predictions)
    adjusted_r2 = 1 - (((1 - r2) * (n - 1)) / (n - p - 1))
    return adjusted_r2
